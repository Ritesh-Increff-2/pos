package com.increff.pos.api;

import java.io.File;
import java.io.IOException;
import java.nio.file.Files;
import java.util.Base64;

import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.stereotype.Service;

import com.increff.pos.dao.InvoiceDao;
import com.increff.pos.db.InvoicePojo;
import com.increff.pos.exception.ApiException;

@Service
public class InvoiceApiImpl implements InvoiceApi {
    @Autowired
    private InvoiceDao invoiceDao;
    
    @Override
    public InvoicePojo addInvoice(InvoicePojo invoicePojo) throws ApiException {
        return invoiceDao.save(invoicePojo);
    }
    @Override
    public String getbase64(String path) throws ApiException {
        try {
            File file = new File(path);
            byte[] bytes = Files.readAllBytes(file.toPath());
            return Base64.getEncoder().encodeToString(bytes);
        } catch (IOException e) {
            throw new ApiException("Error reading file: " + e.getMessage());
        }
    }
}
